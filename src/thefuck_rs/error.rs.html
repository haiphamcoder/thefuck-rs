3/// Main error type for the thefuck-rs application

17    /// Rule-related errors

84/// Error context for better error messages
85pub trait ErrorContext<T> {
86    /// Add context to an error
87    #[allow(clippy::type_complexity)]
88    fn with_context<C>(self, context: C) -> TheFuckResult<T>
89    where
90        C: std::fmt::Display;
91}
92
93impl<T> ErrorContext<T> for TheFuckResult<T> {
94    fn with_context<C>(self, context: C) -> TheFuckResult<T>
95    where
96        C: std::fmt::Display,
97    {
98        self.map_err(|e| match e {
99            TheFuckError::ParseError(msg) => TheFuckError::ParseError(format!("{context}: {msg}")),
100            TheFuckError::ShellExecutionError(msg) => {
101                TheFuckError::ShellExecutionError(format!("{context}: {msg}"))
102            }
103            TheFuckError::RuleExecutionError(msg) => {
104                TheFuckError::RuleExecutionError(format!("{context}: {msg}"))
105            }
106            TheFuckError::ConfigError(msg) => {
107                TheFuckError::ConfigError(format!("{context}: {msg}"))
108            }
109            TheFuckError::FileSystemError(msg) => {
110                TheFuckError::FileSystemError(format!("{context}: {msg}"))
111            }
112            TheFuckError::ProcessError(msg) => {
113                TheFuckError::ProcessError(format!("{context}: {msg}"))
114            }
115            TheFuckError::CommandExecutionError(msg) => {
116                TheFuckError::CommandExecutionError(format!("{context}: {msg}"))
117            }
118            TheFuckError::HistoryError(msg) => {
119                TheFuckError::HistoryError(format!("{context}: {msg}"))
120            }
121            TheFuckError::ValidationError(msg) => {
122                TheFuckError::ValidationError(format!("{context}: {msg}"))
123            }
124            TheFuckError::NetworkError(msg) => {
125                TheFuckError::NetworkError(format!("{context}: {msg}"))
126            }
127            TheFuckError::PathError(msg) => TheFuckError::PathError(format!("{context}: {msg}")),
128            _ => e,
129        })
130    }
131}

180    #[test]
181    fn test_error_creation() {
182        let parse_err = TheFuckError::parse_error("Invalid command syntax");
183        assert!(matches!(parse_err, TheFuckError::ParseError(_)));
184
185        let shell_err = TheFuckError::unsupported_shell("cmd.exe");
186        assert!(matches!(shell_err, TheFuckError::UnsupportedShell(_)));
187
188        let no_rules_err = TheFuckError::no_rules_found("git psh");
189        assert!(matches!(no_rules_err, TheFuckError::NoRulesFound(_)));

192    #[test]
193    fn test_error_context() {
194        #[allow(clippy::type_complexity)]
195        let result: TheFuckResult<()> = Err(TheFuckError::parse_error("test"));
196        let result_with_context = result.with_context("parsing command");
197
198        match result_with_context {
199            Err(TheFuckError::ParseError(msg)) => {
200                assert!(msg.contains("parsing command"));
201                assert!(msg.contains("test"));
202            }
203            _ => panic!("Expected ParseError"),
204        }
205    }
